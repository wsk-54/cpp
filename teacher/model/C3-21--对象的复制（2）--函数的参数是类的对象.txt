#include <iostream.h>

class Time
{
 public:
	Time(int,int,int);

	int hour;
	int minute;
	int sec;
	void get_time( );  

};

Time::Time(int h,int m,int s)
{
	hour=h;
	minute=m;
	sec=s;
}

void Time::get_time( ) {cout<<hour<<":"<<minute<<":" <<sec<<endl;}

void fun(Time t)
{
	t.hour=20;
}
void main()
{
	
	Time t1(10,15,36); 
	t1.get_time( );
	fun(t1);
	t1.get_time( );
}
